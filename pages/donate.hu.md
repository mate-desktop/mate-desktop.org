A MATE asztali környezet teljesen ingyenes és minden érintett önkéntes.

Fontolja meg a MATE projektnek adományozást. Örülünk, ha elismerik a beletett
erőfeszítést. Minden összegyűjtött pénzt a kiszolgálási költségekre, hardverekre
és a lehetséges utazási költségekre fordítunk, hogy a MATE csapat tagjai
releváns konferenciákon vehessenek részt.

Nem kérünk sokat, és Önre bízzuk az összeget.

  function validateMonthlyForm() {
    var n = document.forms["monthly"]["other"].value;
      if (n) {
        if (!isNaN(parseFloat(n)) && isFinite(n) && (n > 0)) {
          document.forms["monthly"]["a3"].value = n;
          return true;
        } else {
          alert("Adjon meg egy érvényes adományozási mennyiséget - köszönjük!");

        return true;
      }
  }

  function validateSingleForm() {
    var n = document.forms["single"]["other"].value;

        return true;
      }
  }
